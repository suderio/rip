19  public class RipServerTest {
20    @BeforeClass
21    public static void setUpBeforeClass() throws Exception {
22      RestAssured.port = 7777;
23      localhost().get("/test").respond("Ok");
24      localhost().post("/test").contains("teste1")
25          .respond(withFile("/teste1.json"));
26      localhost().post("/test").contains("teste2").and().contains("something")
27          .respond(withFile("/teste2.json"));
28      localhost().put("/test").containsAll("123", "456").respond("123456");
29      localhost().put("/test").containsAny("789", "987").respond("789987");
30      localhost().put("/test").respond("Ok");
31      localhost().delete("/test").contains("xpto").or().contains("abcd")
32          .respond("Ok", SC_OK);
33      localhost().delete("/test").respond("Not Ok", SC_FORBIDDEN);
34    }

36    @AfterClass
37    public static void tearDownAfterClass() throws Exception {
38      // stop(7777);
39    }

41    @Test
42    public void testDelete() {
43      given().body("something abcd something...").when().delete("/test").then()
44          .statusCode(SC_OK);
45      given().body("something xpto something...").when().delete("/test").then()
46          .statusCode(SC_OK);
47      given().body("something abcd xpto something...").when().delete("/test")
48          .then().statusCode(SC_OK);
49      when().delete("/test").then().statusCode(SC_FORBIDDEN);
50    }
51  
52    @Test
53    public void testDeterministic() {
54      IntStream.rangeClosed(1, 1000).boxed().parallel().forEach(i -> {
55        given().body("something abcd xpto something...").when().delete("/test")
56            .then().statusCode(SC_OK);
57        given().body("something 987 something 789...").when().put("/test").then()
58            .content(containsString("789987"));
59        given().body("teste1").when().post("/test").then()
60            .content(containsString("Ok"));
61        when().get("/test").then().content(containsString("Ok"));
62      });
63    }
