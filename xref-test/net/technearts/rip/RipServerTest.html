21  public class RipServerTest {
22      @BeforeClass
23      public static void setUpBeforeClass() throws Exception {
24          RestAssured.port = 8888;
25          localhost(8888).get("/test").respond("Ok");
26          localhost(8888).post("/test").contains("teste1").respond(withFile("/teste1.json"));
27          localhost(8888).post("/test").contains("teste2").and().contains("something").respond(withFile("/teste2.json"));
28          localhost(8888).put("/test").containsAll("123", "456").respond("123456");
29          localhost(8888).put("/test").containsAny("789", "987").respond("789987");
30          localhost(8888).put("/test").respond("Ok");
31          localhost(8888).delete("/test").contains("xpto").or().contains("abcd").respond("Ok", HttpStatus.SC_OK);
32          localhost(8888).delete("/test").respond("Not Ok", HttpStatus.SC_FORBIDDEN);
33      }
34  
35      @AfterClass
36      public static void tearDownAfterClass() throws Exception {
37          stop(8888);
38      }

40      @Test
41      public void testGet() {
42              when().get("/test").then().content(containsString("Ok"));
43      }
44  
45      @Test
46      public void testPost() {
47              when().get("/test").then().content(containsString("Ok"));
48              given().body("teste1").when().post("/test").then().content(containsString("Ok"));
49              given().body("teste2 something xpto").when().post("/test").then().content(containsString("KO"));
50      }
51  
52      @Test
53      public void testPut() {
54              given().body("something 123 something 456...").when().put("/test").then().content(containsString("123456"));
55              given().body("something 987 something 789...").when().put("/test").then().content(containsString("789987"));
56              when().put("/test").then().content(containsString("Ok"));
57      }
58  
59      @Test
60      public void testDelete() {
61              given().body("something abcd something...").when().delete("/test").then().statusCode(HttpStatus.SC_OK);
62              given().body("something xpto something...").when().delete("/test").then().statusCode(HttpStatus.SC_OK);
63              given().body("something abcd xpto something...").when().delete("/test").then().statusCode(HttpStatus.SC_OK);
64              when().delete("/test").then().statusCode(HttpStatus.SC_FORBIDDEN);
65      }
66      
67      @Test
68      public void testDeterministic() {
69          IntStream.rangeClosed(1, 1000).boxed().parallel().forEach(i -> {
70              given().body("something abcd xpto something...").when().delete("/test").then().statusCode(HttpStatus.SC_OK);
71              given().body("something 987 something 789...").when().put("/test").then().content(containsString("789987"));
72              given().body("teste1").when().post("/test").then().content(containsString("Ok"));
73              when().get("/test").then().content(containsString("Ok"));
74          });
75      }
